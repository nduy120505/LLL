  // flatArray.eq(g) tráº£ vá» máº£ng 0 vÃ  1. reduce(sum) Ä‘á»ƒ Ä‘áº¿m sá»‘ lÆ°á»£ng sá»‘ 1.
  var count = flatArray.eq(g)
              .reduce(ee.Reducer.sum(), [0])
              .get([0]);
  return count;
});

print('Histogram', histList);

// --- BÆ°á»›c 3: PDF ---
var pdfList = histList.map(function(cnt) {
  return ee.Number(cnt).divide(N);
});

// --- BÆ°á»›c 4: CDF (Cá»™ng dá»“n) ---
var cdfList = ee.List(pdfList).iterate(
  function(val, acc) {
    acc = ee.List(acc);
    val = ee.Number(val);
    var cumPrev = ee.Number(acc.get(-1));
    var cumNew = cumPrev.add(val);
    return acc.add(cumNew);
  },
  ee.List([0])
);
// Bá» pháº§n tá»­ 0 khá»Ÿi táº¡o
cdfList = ee.List(cdfList).slice(1);

// --- BÆ°á»›c 5: Báº£ng Ã¡nh xáº¡ s = round((L-1)*CDF) ---
// index lÃ  má»©c xÃ¡m cÅ©, value lÃ  má»©c xÃ¡m má»›i
var mappingList = cdfList.map(function(cdf) {
  return ee.Number(cdf).multiply(L - 1).round().int(); 
});

print('Báº£ng Ã¡nh xáº¡ (Mapping List)', mappingList);

// --- BÆ°á»›c 6: Ãp dá»¥ng Ã¡nh xáº¡ cho áº£nh I ---

//  DÃ¹ng flatList (lÃ  ee.List) Ä‘á»ƒ map
//  má»—i pixel cÅ© (v), láº¥y giÃ¡ trá»‹ táº¡i vá»‹ trÃ­ v trong mappingList

var I_eq_flat_list = flatList.map(function(v) {
  v = ee.Number(v).int(); // Äáº£m báº£o index lÃ  sá»‘ nguyÃªn
  return mappingList.get(v); // Láº¥y giÃ¡ trá»‹ má»›i tá»« báº£ng Ã¡nh xáº¡
});

// --- BÆ°á»›c 7: Chuyá»ƒn List trá»Ÿ láº¡i Ma tráº­n 8x8 ---
